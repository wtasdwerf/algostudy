## 문제

> 가로 10줄, 세로 10줄에 1 또는 0이 적혀진 배열이 있다. 이러한 배열 안에 있는 숫자 1들이 만드는 모양이 한 개의 직각이등변삼각형인지 아닌지 알아내는 프로그램을 작성하시오.
> 

직각이등변삼각형의 적어도 한 변은 수평선 또는 수직선이다. 단, 직각이등변삼각형의 내부도 1로 채워져 있어야 한다. 입력된 모양은 삼각형이 아닐 수 있다.

def check(lst, posR, posC, idx, sz):
    for r in range(10):
        for c in range(10):
            if 0 <= r+posR < 10 and 0 <= c+posC < 10:
                if triangle[idx][r][c] != 0 and triangle[idx][r][c] <= sz:
                    if lst[r+posR][c+posC] == 0:
                        return False
                else:
                    if lst[r+posR][c+posC] == 1:
                        return False
            else:
                if triangle[idx][r][c] != 0 and triangle[idx][r][c] <= sz:
                    return False
    for r in range(10):
        for c in range(10):
            if posR <= r < posR+10 and posC <= c < posC+10:
                continue
            if lst[r][c]:
                return False

    return True

def solve(lst):
    for posR in range(-9, 10):
        for posC in range(-9, 10):
            for idx in range(0, 8):
                for sz in range(2, 11):

                        return ret

    return []
