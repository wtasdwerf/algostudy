## 풀이

```python
T = int(input())
for tc in range(1, T+1):
    K, N, M = map(int, input().split())
    # 충전소
    station = [False for i in range(101)]
    # 각 위치마다 주유소 존재 여부
    station_list = list(map(int, input().split()))

    for s in station_list:
        station[s] = True

    # 역추적
    cur = N
    ans = 0
    while cur > 0:
        # 남은 정류장이 K개 이하이면 종점 도착이 가능한 케이스를 판단할 수 있음과 동시에
        # 더이상 충전할 장소가 없다는것도 알 수 있음 즉, 현재 ans가 답
        if cur <= K:
            cur = 0
            continue

        target = cur # 다음 이동 위치
        # 들려야하는 충전소 찾기 (역추적)
        for i in range(1, K+1):
            if station[cur-i]:
                target = cur-i
        
        # 찾지 못했으면 불가능한 케이스
        if target == cur:
            ans = 0
            break
        
        # 충전소를 찾으면 해당 위치로 이동
        cur = target
        ans += 1

    print(f'#{tc} {ans}')
```
